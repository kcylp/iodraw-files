        PowerNet[("本安电源<br/>KDK660等")] %% Renamed to avoid conflict
        KJ95X_Data(["现有KJ95X<br/>数据接口"])
    end

    SensorsA -- 信号 --> PLC
    SensorsD -- 信号 --> Substation
    PLC -- 数据 --> Switch
    Substation -- 数据 --> Switch
    KJ95X_Data -- 数据 --> Switch
    Switch -- 数据 --> IPC

    %% Control Flows
    Operator -- 操作 --> SCADA
    IPC -- 运行 --> SCADA
    SCADA -- 控制指令 --> Switch
    Switch -- 控制指令 --> PLC
    PLC -- 控制 --> Actuators

    %% Power Supply Connections
    PowerField -- 供电 --> PLC
    PowerField -- 供电 --> Substation
    PowerNet -- 供电 --> Switch %% Assuming central switch power, field power distinct or same
